# Оракул — расписание cron (рутина §27). АДАПТИРОВАНО ПОД ЭТУ МАШИНУ (Нед.9):
#   • cron НЕ наследует ни venv, ни .env → КАЖДАЯ команда сама поднимает окружение:
#         cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python ...
#     - `cd /root/oracle` — рабочая директория (относительные пути storage/journal/.env);
#     - `set -a && . ./.env && set +a` — экспорт ВСЕХ ключей (.env = строки export KEY=val):
#         OPENROUTER_API_KEY (LLM), EODHD_API_KEY (котировки), NEWSAPI_AI_KEY (новости);
#     - АБСОЛЮТНЫЙ /root/oracle/.venv/bin/python — голого `python` на PATH НЕТ,
#       /usr/bin/python3 зависимостей НЕ видит.
#   • Каждая строка проверена ручным запуском перед установкой (журнал Нед.9) + проверка
#     срабатывания временной строкой с записью в journal/cron.log (Нед.9, п.4).
# Рутина идёт НАПРЯМУЮ через python — дешевле и воспроизводимо. Claude Code — для разработки/разбора.
#
# Установка:  crontab /root/oracle/ops/crontab.txt
# Проверка:   crontab -l
# ════════════════════════════════════════════════════════════════════════════════

SHELL=/bin/bash
MAILTO=""

# Суточный новостной сбор (GDELT + NewsAPI.ai + pytrends → нормализация/теги/дедуп): ежедневно 06:45
45 6 * * *   cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python data/news_ingest.py >> journal/cron.log 2>&1

# Суточная дозагрузка котировок ядра + Tier 0 (фундаментал/инсайдеры/календарь) + опционы
# (Unicorn Bay IV/OI/skew) из EODHD в storage/oracle.db: 06:50. --extras/--options освежают
# флоат/short%float/IV ежедневно (иначе устаревают). Нужен EODHD_API_KEY → source .env.
50 6 * * *   cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python data/eodhd.py --extras --options >> journal/cron.log 2>&1

# Калибровочный режим §17.3 (мелкие недельные §9-прогнозы, запечатывание): вторник и пятница 08:00
0 8 * * 2,5  cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python orchestrator/run.py --mode calibrate >> journal/cron.log 2>&1

# Тематический режим Brent §17.2 (полный цикл по теме + запечатанный прогноз): будни 07:30
30 7 * * 1-5 cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python orchestrator/run.py --mode theme --asset brent >> journal/cron.log 2>&1

# Сверка исходов §10.10 (дозревшие прогнозы → close из oracle.db → outcomes.jsonl): ежедневно 21:00
0 21 * * *   cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python orchestrator/run.py --mode resolve >> journal/cron.log 2>&1

# Боевой прогон полной воронки §6 (свободная генерация идей): понедельник 09:00
0 9 * * 1    cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python orchestrator/run.py --mode funnel >> journal/cron.log 2>&1

# Панель бюджета §15: пересборка каждое утро 07:00; код выхода 3 = превышение потолка limits.yaml
0 7 * * *    cd /root/oracle && set -a && . ./.env && set +a && /root/oracle/.venv/bin/python ops/budget.py >> journal/cron.log 2>&1

# Напоминание о еженедельном разборе (сам разбор — в Claude Code: /review-week): пятница 18:00
0 18 * * 5   cd /root/oracle && echo "$(date): пора /review-week в Claude Code" >> journal/reminders.log
